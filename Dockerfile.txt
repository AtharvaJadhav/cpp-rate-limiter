FROM ubuntu:22.04

# Install dependencies
RUN apt-get update && apt-get install -y \
    build-essential \
    cmake \
    git \
    pkg-config \
    libhiredis-dev \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Install redis-plus-plus
RUN git clone https://github.com/sewenew/redis-plus-plus.git /tmp/redis-plus-plus && \
    cd /tmp/redis-plus-plus && \
    mkdir build && cd build && \
    cmake -DCMAKE_BUILD_TYPE=Release .. && \
    make -j$(nproc) && \
    make install && \
    ldconfig && \
    rm -rf /tmp/redis-plus-plus

# Set working directory
WORKDIR /app

# Copy source code
COPY main.cpp CMakeLists.txt ./

# Create third_party directory and download dependencies
RUN mkdir -p third_party/httplib third_party/json

# Download httplib
RUN curl -L "https://raw.githubusercontent.com/yhirose/cpp-httplib/v0.14.1/httplib.h" \
    -o third_party/httplib/httplib.h

# Download nlohmann/json
RUN git clone --depth 1 --branch v3.11.2 https://github.com/nlohmann/json.git third_party/json

# Build the application
RUN mkdir build && cd build && \
    cmake -DCMAKE_BUILD_TYPE=Release .. && \
    make -j$(nproc)

# Expose port
EXPOSE 8080

# Set environment variables
ENV REDIS_URL=tcp://redis:6379
ENV PORT=8080

# Run the application
CMD ["./build/rate_limiter"]
